create table agencias
(        
        num_agencia NUMERIC(4) not null, 
        endereco VARCHAR(45),
        constraint pk_agencias_num PRIMARY KEY (num_agencia)
);

create table usuarios
(
	id integer not null generated always as identity (start with 1, increment by 1),
	nome VARCHAR(45) not null,
	CPF varchar(14) unique,
	telefone varchar(11),
	endereco varchar(45),
	senha varchar(6),
        constraint pk_usuarios_id PRIMARY KEY (id)
);

create table tipos_conta (
        id integer not null generated always as identity (start with 1, increment by 1),
	descricao varchar(25) unique,
        constraint tipos_conta_pk PRIMARY KEY (id)
);

create table Contas
(
        num_agencia numeric(4),	
        num_conta numeric(5),
        id_tipo_conta integer, 
        id_usuario integer,                    	
	saldo NUMERIC(12,2) default 0,	
        constraint contas_pk PRIMARY KEY (num_agencia,num_conta,id_tipo_conta,id_usuario),
        foreign key (id_usuario) references usuarios (id),
        foreign key (num_agencia) references agencias (num_agencia),
        foreign key (id_tipo_conta) references tipos_conta (id)
);

create table tipos_operacao(
        id integer not null generated always as identity (start with 1, increment by 1),
	descricao varchar(25) unique,
        constraint tipos_operacao_pk PRIMARY KEY (id)

);

create table log_operacoes(        
	id integer not null generated always as identity (start with 1, increment by 1),
	id_tipo_operacao INTEGER,
	num_agencia numeric(4),
        num_conta numeric(5),
        id_tipo_conta integer,
        id_usuario integer,
	valor NUMERIC(12,2),
	data TIMESTAMP,
        constraint operacoes_pk PRIMARY KEY (id),
        foreign key (id_tipo_operacao) references tipos_operacao (id),
        foreign key (num_agencia, num_conta,id_tipo_conta,id_usuario) references contas (num_agencia, num_conta,id_tipo_conta,id_usuario)
             
);

CREATE VIEW login
	(NUM_AGENCIA, NUM_CONTA,id_tipo_conta,NOME,id_usuario) 
	AS SELECT contas.NUM_AGENCIA, contas.NUM_CONTA, contas.id_tipo_conta, usuarios.NOME, usuarios.id
FROM contas JOIN usuarios 
ON contas.ID_usuario = usuarios.id;
